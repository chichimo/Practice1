	@Test
	@Parameters({"username","password","fname","mname","lname","Email", "pNo","comment"})
	public void login(String UserID, String pwd, String fname, String mname, String lname, String Email, String pno, String Comment) {
		System.setProperty("webdriver.gecko.driver", "C://Users//hashi//Downloads//Drivers//geckodriver.exe");
		WebDriver driver = new ChromeDriver();
		driver.get("http://opensource.demo.orangehrmlive.com");
		driver.findElement(By.id("txtUsername")).sendKeys(UserID);
		driver.findElement(By.id("txtPassword")).sendKeys(pwd);
		driver.findElement(By.name("Submit")).click();
		driver.findElement(By.linkText("Recruitment")).click();
		driver.findElement(By.id("btnAdd")).click();
		driver.findElement(By.xpath("//*[@id=\"addCandidate_firstName\"]")).sendKeys(fname);
		driver.findElement(By.xpath("//*[@id=\"addCandidate_middleName\"]")).sendKeys(mname);
		driver.findElement(By.xpath("//*[@id=\"addCandidate_lastName\"]")).sendKeys(lname);
		driver.findElement(By.xpath("//*[@id=\"addCandidate_email\"]")).sendKeys(Email);
		driver.findElement(By.xpath("//*[@id=\"addCandidate_contactNo\"]")).sendKeys(pno);
		driver.findElement(By.xpath("//*[@id=\"addCandidate_comment\"]")).sendKeys(Comment);
		driver.findElement(By.id("btnSave")).click();
		
	}

}
